                class TaskManager implements TaskManagerInterface {

                        return tasks;

                    @Override
                    public void addTask(TaskInterface task) {
                        tasks.add(task);
                    }

                    public void viewAllTasks() {
                        tasks.forEach(System.out::println);
                    }
                
                    @Override
                    public void updateTaskStatus(String title, TaskStatus newStatus) {
                        tasks.stream()
                                .filter(task -> task.getTitle().equalsIgnoreCase(title))
                                .forEach(task -> task.setStatus(newStatus));
                    }
                
                    @Override
                    public void deleteTask(String title) {
                        tasks.removeIf(task -> task.getTitle().equalsIgnoreCase(title));
                    }
                
                    @Override
                    public List filterTasksByStatus(TaskStatus status) {
                        return tasks.stream()
                                .filter(task -> task.getStatus() == status)
                                .collect(Collectors.toList());
                    }
                }
