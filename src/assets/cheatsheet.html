        public class LibrarySystem {
            private List bookTitles;
            
            // Constructor to initialize the list of book titles
            public LibrarySystem() {
                bookTitles = new ArrayList<>();
                bookTitles.add("To Kill a Mockingbird");
                bookTitles.add("1984");
                bookTitles.add("The Great Gatsby");
                 
            }
            
            public boolean isBookAvailable(String userQuery) {
                for (String title : bookTitles) {
                    if (title.compareToIgnoreCase(userQuery) == 0) {
                        return true;
                    }
                }
                return false;
            }
            
            public static void main(String[] args) {
                LibrarySystem library = new LibrarySystem();
            
                // Simulate a user search
                String userSearch = "tO kILL A mOCKINGBIRD";
            
                if (library.isBookAvailable(userSearch)) {
                    System.out.println("Book " + userSearch + " available!");
                } else {
                    System.out.println("Book " + userSearch + "not available.");
                }
            }
            output:The book 'tO kILL A mOCKINGBIRD' is available!

        }
