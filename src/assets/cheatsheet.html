import java.util.Objects;

public class Contact {
    private String name;
    private String email;

    public Contact(String name, String email) {
        this.name = name;
        this.email = email;
    }

    public String getEmail() {
        return email;
    }

    // Override the equals method to determine equality based on the email field.
    @Override
    public boolean equals(Object obj) {
        if (this == obj) {
            return true;
        }
        if (obj == null || getClass() != obj.getClass()) {
            return false;
        }
        Contact contact = (Contact) obj;
        return Objects.equals(email, contact.email);
    }

    // hashCode method should also be overridden when equals is overridden.
    @Override
    public int hashCode() {
        return Objects.hash(email);
    }

    public static void main(String[] args) {
        Contact aliceContact = new Contact("Alice", "alice@example.com");
        Contact bobContact = new Contact("Bob", "bob@example.com");
        Contact anotherAliceContact = new Contact("Alice B.", "alice@example.com");

        // Should print "Duplicate contact found!" because the emails are the same.
        if (aliceContact.equals(anotherAliceContact)) {
            System.out.println("Duplicate contact found!");
        }

        // Should not print anything because the emails are different.
        if (aliceContact.equals(bobContact)) {
            System.out.println("This won't be printed because the contacts are different.");
        }
    }
}
Note:In this script, equals is overridden in the Contact class to ensure that two Contact
    
objects are considered equal if their emails are the same. The hashCode method is also overridden
    
because it's good practice to override hashCode when equals is overridden to maintain the contract
    
between these two methods, which is essential for certain collections like HashSet or HashMap to 
    
operate correctly.

    public boolean equals(Object obj) {

            return true;

        if (obj == null || getClass() != obj.getClass()) {

public class StringInternExample {

    public static void main(String[] args) {
        // Two strings created in a way they don't refer to the same memory initially.
        String str1 = new String("Hello");
        String str2 = new String("Hello");

        System.out.println(str1 == str2);  // Outputs: false

        // Interning the strings
        str1 = str1.intern();
        str2 = str2.intern();

        System.out.println(str1 == str2);  // Outputs: true
    }
}
/*
1> Initially, str1 and str2 point to different memory locations. 
So, the comparison str1 == str2 returns false.
2> After interning both strings using intern(),
they both point to the same memory location in the string pool.
Thus, the comparison str1 == str2 returns true.
*/

public class FileUtility {

    public static String getFileExtension(String filename) {
        int lastDotPosition = filename.lastIndexOf('.');
        
        if (lastDotPosition != -1 && lastDotPosition < filename.length() - 1) {
            return filename.substring(lastDotPosition + 1);
        }
        
        return null;  // No valid extension found
    }

    public static void main(String[] args) {
        String f1 = "document.txt";
        String f2 = "archive.tar.gz";
        
        System.out.println("File extension of '" + f1 + "' is: " + getFileExtension(f1));
        System.out.println("File extension of '" + f2 + "' is: " + getFileExtension(f2));
    }
}
//File extension of 'document.txt' is: txt
//File extension of 'archive.tar.gz' is: gz

public class DomainUtility {

    public static String getMainDomain(String fullDomain) {
        int lastDotPosition = fullDomain.lastIndexOf('.');
        
        if (lastDotPosition > 0) {
            int secondLastDotPosition = fullDomain.lastIndexOf('.', lastDotPosition - 1);
            
            if (secondLastDotPosition != -1 && secondLastDotPosition < lastDotPosition - 1) {
                return fullDomain.substring(secondLastDotPosition + 1, lastDotPosition);
            }
        }
        
        return null;  // No valid main domain found
    }

    public static void main(String[] args) {
        String domain1 = "mail.google.com";
        String domain2 = "archive.sub.domain.com";
        
        System.out.println("Main domain of '" + domain1 + "' is: " + getMainDomain(domain1));
        System.out.println("Main domain of '" + domain2 + "' is: " + getMainDomain(domain2));
    }
}
